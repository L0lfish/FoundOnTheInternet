// Avant de pouvoir créer cette classe de connexion il faut installer le driver jdbc mysql ! 




package nomPackage;								// Le nom du package dans lequel se trouve la classe

import java.sql.Connection;							// Import des classes sql qui nous serviront pour la 
import java.sql.DriverManager;
import java.sql.SQLException;


public class Connexion {							// Création de la classe de connexion. Attention, ne pas utiliser "Connection" qui est une classe déjà existante (que nous utilisons d'ailleurs plus bas)

    static Connection conex;							// On crée une connexion

    static {
        try {
            Class.forName("com.mysql.jdbc.Driver");				// On appelle la classe du driver
            String url = "jdbc:mysql://localhost:3306/databaseName";		// URL de la database
            String user = "root";						// On définit l'username et le password
            String password = "";
            conex = DriverManager.getConnection(url, user, password);		// On appelle ici la classe drivermanager du jdbc
          
        } catch (ClassNotFoundException | SQLException e) {			// Sinon on catch l'exception
         
        }
    }

public static Connection getConnexion(){					// On crée le getter de la connexion.
    return conex;
}

}
